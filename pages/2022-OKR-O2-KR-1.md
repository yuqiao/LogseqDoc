- query-table:: true
  #+BEGIN_QUERY
  {:title "KR内容"
   :query [:find (pull ?b [*])
           :in $ ?current-page
           :where
           [?p :page/name ?current-page]
           [?b :block/ref-pages ?p]
           [?b :block/page ?bp]
           [?bp :page/name ?bp-name]
           [(not= "okr" ?bp-name)]
           (not [?b :block/marker ?marker]
           [(contains? #{"TODO" "DOING" "DONE" "LATER" "NOW"} ?marker)])]
   :inputs [:current-page]}
  #+END_QUERY
